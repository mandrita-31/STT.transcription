const startButton = document.getElementById('start');
const stopButton = document.getElementById('stop');

if (!('webkitSpeechRecognition' in window)) {
    alert('Your browser does not support the Web Speech API. Please use Chrome or another compatible browser.');

    const recognition = new webkitSpeechRecognition();
    recognition.lang = 'en-US';
    recognition.interimResults = true;
    recognition.continuous = true;

    let transcript = '';
    let interimTranscript = '';
    let lastResultTime = Date.now(); // Initialize with the current time

    recognition.onresult = (event) => {
        const currentTime = Date.now();
        const pauseDuration = (currentTime - lastResultTime) / 1000; // Pause duration in seconds

            } else {

        // Apply punctuation based on pause duration
        const textWithPunctuation = addPunctuation(transcript + interimTranscript, pauseDuration);
        transcriptionDiv.innerText = textWithPunctuation;
    };

    recognition.onstart = () => {
        statusDiv.innerText = 'Status: Listening...';
        startButton.disabled = true;
        stopButton.disabled = false;

function addPunctuation(text, pauseDuration) {
    // Handle punctuation based on pause duration
    if (pauseDuration > 1.5) {
        return text + '. '; // Add period if the pause is longer than 1.5 seconds
    } else if (pauseDuration > 1) {
        return text + ', '; // Add comma if the pause is longer than 1 second
    } else {
        return text; // No punctuation for shorter pauses
    }
}

    • const currentTime = Date.now();: Gets the current timestamp when a new result is received.
    • const pauseDuration = (currentTime - lastResultTime) / 1000;: Calculates the pause duration in seconds by subtracting the last result time from the current time.

  2. Adding Punctuation:

    • addPunctuation(transcript + interimTranscript, pauseDuration);: Calls the addPunctuation function with the combined transcript and interim text, as well as the pause duration.

  3. Punctuation Logic:

    • if (pauseDuration > 1.5) { return text + '. '; }: Adds a period if the pause is longer than 1.5 seconds.
    • else if (pauseDuration > 1) { return text + ', '; }: Adds a comma if the pause is longer than 1 second but less than or equal to 1.5 seconds.
